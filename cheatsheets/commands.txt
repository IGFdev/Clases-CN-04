----- EVERYTHING -----
- Delete all images, containers and volumes
    docker system prune -a

- Run docker compose
    docker-compose up

- Stop docker compose
    docker-compose down
    docker-compose --rmi all -v

----- IMAGES -----

- Build new image from dockerfile   
    docker build -t myApp .  
    docker build -t myApp:v2 .
    docker build -t myapp:nodemon .

- List images
    docker images

- Delete image  
    docker image rm myApp
    docker image rm myApp -f

----- CONTAINERS -----

- List containers
    docker ps
    docker ps -a

- Create container
    docker run --name myapp_c -p 4000:4000 -d myapp:v1
    docker run --name myapp_c_nodemon -p 4000:4000 --rm myapp:nodemon
    docker run --name myapp_c_nodemon -p 4000:4000 --rm -v N:\Workspaces\NodeJS\docker-course\app:/app -v /app/node_modules myapp:nodemon

- Stop container
    docker stop myapp_c

- Start existing container
    docker start myapp_c

- Delete container
    docker container rm myapp_c1
    docker container rm myapp_c2 myapp_c3
